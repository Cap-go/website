  </Card>
  <Card title="Rapid setup" icon="rocket">
    Ship fully native payment sheets without manually wiring platform-specific code.
  </Card>
  <Card title="Tokenized payments" icon="shield">
    Receive payment tokens ready for your gateway or PSP integrations.
  </Card>
  <Card title="Customizable flows" icon="chart">
    Configure line items, contact fields, and billing requirements to fit your checkout.
  </Card>
</CardGrid>
