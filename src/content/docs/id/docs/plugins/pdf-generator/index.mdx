import { Card, CardGrid } from '@astrojs/starlight/components';

Konversi HTML atau halaman jarak jauh menjadi PDF yang dapat dibagikan dengan renderer native Capgo untuk iOS dan Android.

<CardGrid stagger>
  <Card title="HTML ke PDF" icon="pencil">
    Masukkan template HTML dan dapatkan dokumen PDF siap dibagikan dalam hitungan detik.
  </Card>
  <Card title="Tangkapan jarak jauh" icon="rocket">
    Tangkap snapshot dari URL yang dapat diakses dan simpan atau distribusikan PDF yang dihasilkan.
  </Card>
  <Card title="Output fleksibel" icon="star">
    Kembalikan data base64 atau aktifkan sheet berbagi native tergantung alur kerja Anda.
  </Card>
  <Card title="Kontrol dokumen" icon="setting">
    Tentukan orientasi, ukuran halaman, dan nama file untuk setiap render.
  </Card>
</CardGrid>

Kunjungi panduan memulai untuk catatan platform dan contoh penggunaan.
